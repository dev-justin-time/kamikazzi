                    const size = box.getSize(new THREE.Vector3());
                    const maxDim = Math.max(size.x || 1, size.y || 1, size.z || 1);
                    const scaleFactor = maxDim > 0 ? 2 / maxDim : 1;
                    obj.scale.set(scaleFactor, scaleFactor, scaleFactor);

                    // Adjust so object's lowest point sits on floor (y=0)
                    const scaledBoxObj = new THREE.Box3().setFromObject(obj);
                    const minYObj = scaledBoxObj.min.y;
                    if (isFinite(minYObj)) {
                      obj.position.y -= minYObj;
                    }

                    // Rotate object to face the camera horizontally
                    if (this.camera) {
                      const lookAtPos = this.camera.position.clone();
                      lookAtPos.y = obj.position.y; // project camera to model height so only yaw is applied
                      obj.lookAt(lookAtPos);
                    }

                    this.scene.add(obj);
                    // clear animations context
                    this.mixer = null; this.actions = {}; this.animations = [];
                    this.selectObject(obj);
                    resolve();
                }, undefined, (err) => {
                    try { URL.revokeObjectURL(url); } catch (e) {}
                    reject(err);
